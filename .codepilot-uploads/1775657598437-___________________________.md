### 2.3 组件实现要点（基于项目样式体系）

```
// components/ads/AdSlot.tsx
import { cn } from "@/lib/utils"; // 项目通用样式拼接工具
import { useEffect, useRef, useState } from "react";

interface AdSlotProps {
  slot: AdSlot
  fallback?: React.ReactNode  // 无广告时的兜底内容
  adContent?: AdContent
}

export function AdSlot({ slot, fallback, adContent }: AdSlotProps) {
  const [isLoaded, setIsLoaded] = useState(false);
  const slotRef = useRef<HTMLDivElement>(null);
  
  // 懒加载逻辑
  useEffect(() => {
    if (!slot.lazyLoad || !slotRef.current) return;
    
    const observer = new IntersectionObserver((entries) => {
      if (entries[0].isIntersecting) {
        setIsLoaded(true);
        observer.disconnect();
      }
    });
    
    observer.observe(slotRef.current);
    return () => observer.disconnect();
  }, [slot.lazyLoad]);

  // 无广告内容时显示兜底
  if (!adContent && !isLoaded) {
    return <AdSkeleton slot={slot} />;
  }
  
  if (!adContent) {
    return <div className="ad-fallback">{fallback}</div>;
  }

  return (
    <div
      ref={slotRef}
      className={cn(
        "ad-slot relative overflow-hidden transition-all duration-300",
        // 复用项目预设样式
        "border border-border rounded-lg",
        slot.type === 'banner' ? "w-full" : "max-w-full",
        slot.type === 'splash' ? "fixed inset-0 z-50" : "",
        slot.type === 'native' ? "bg-background" : "bg-card",
        slot.type === 'insertion' ? "card-custom" : "shadow-card hover:shadow-hover"
      )}
      data-slot-id={slot.id}
      style={slot.size ? { 
        width: slot.size.width + 'px', 
        height: slot.size.height + 'px' 
      } : {}}
    >
      {/* 广告内容容器 */}
      <a 
        href={adContent.targetUrl} 
        className="block w-full h-full"
        onClick={(e) => trackAdClick(adContent.id, slot.id)}
      >
        {/* 广告图片 */}
        <div className="relative w-full h-[60%] overflow-hidden rounded-t-lg">
          <img 
            src={adContent.imageUrl} 
            alt={adContent.title}
            className="w-full h-full object-cover transition-transform hover:scale-105"
            loading="lazy"
          />
        </div>
        
        {/* 广告文案 */}
        <div className="p-4">
          <h3 className="font-semibold text-foreground mb-1">{adContent.title}</h3>
          <p className="text-sm text-muted-foreground mb-3 line-clamp-2">
            {adContent.description}
          </p>
          
          {/* CTA 按钮 - 复用项目按钮样式 */}
          <button 
            className="btn btn-primary w-full"
            onClick={(e) => e.stopPropagation()}
          >
            {adContent.ctaText}
          </button>
        </div>
      </a>
    </div>
  );
}

// 广告骨架屏组件
function AdSkeleton({ slot }: { slot: AdSlot }) {
  return (
    <div 
      className={cn(
        "ad-skeleton animate-pulse border border-border rounded-lg",
        slot.size ? `w-[${slot.size.width}px] h-[${slot.size.height}px]` : "w-full h-40",
        slot.type === 'native' ? "bg-background" : "bg-card"
      )}
    >
      <div className="w-full h-2/3 bg-muted rounded-t-lg"></div>
      <div className="p-4">
        <div className="h-4 bg-muted-foreground/20 rounded w-3/4 mb-2"></div>
        <div className="h-3 bg-muted-foreground/20 rounded w-1/2 mb-4"></div>
        <div className="h-9 bg-muted-foreground/20 rounded w-full"></div>
      </div>
    </div>
  );
}

// 点击追踪函数
function trackAdClick(adId: string, slotId: string) {
  // 上报逻辑
  console.log(`Ad clicked: ${adId} (slot: ${slotId})`);
}
```

```
// components/download/DownloadCountdown.tsx
import { useState, useEffect } from "react";
import { cn } from "@/lib/utils";

interface DownloadCountdownProps {
  seconds: number;
  downloadUrl: string;
  softwareName: string;
}

export default function DownloadCountdown({ seconds, downloadUrl, softwareName }: DownloadCountdownProps) {
  const [remaining, setRemaining] = useState(seconds);
  const [canDownload, setCanDownload] = useState(false);

  useEffect(() => {
    if (remaining <= 0) {
      setCanDownload(true);
      return;
    }

    const timer = setInterval(() => {
      setRemaining(prev => prev - 1);
    }, 1000);

    return () => clearInterval(timer);
  }, [remaining]);

  return (
    <div className="w-full max-w-xs mt-4">
      {/* 倒计时圆环 */}
      <div className="relative w-20 h-20 mx-auto mb-4">
        <svg className="w-full h-full" viewBox="0 0 100 100">
          <circle
            cx="50"
            cy="50"
            r="45"
            fill="none"
            stroke="rgb(var(--muted))"
            strokeWidth="4"
          />
          <circle
            cx="50"
            cy="50"
            r="45"
            fill="none"
            stroke="rgb(var(--accent))"
            strokeWidth="4"
            strokeDasharray={283}
            strokeDashoffset={283 * (1 - remaining / seconds)}
            strokeLinecap="round"
            transform="rotate(-90 50 50)"
            className="transition-all duration-1000"
          />
          <text
            x="50"
            y="55"
            textAnchor="middle"
            fontSize="16"
            fontWeight="bold"
            fill="rgb(var(--foreground))"
          >
            {remaining}
          </text>
        </svg>
      </div>

      {/* 提示文本 */}
      <p className="text-center text-muted-foreground mb-4">
        {canDownload ? '下载已准备就绪' : `即将开始下载，请等待 ${remaining} 秒`}
      </p>

      {/* 下载按钮 - 复用项目按钮样式 */}
      <a
        href={downloadUrl}
        className={cn(
          "admin-primary-btn w-full justify-center",
          !canDownload && "opacity-60 cursor-not-allowed pointer-events-none"
        )}
        disabled={!canDownload}
      >
        {canDownload ? `立即下载 ${softwareName}` : '跳过广告'}
      </a>
    </div>
  );
}
```

```
model Software {
  id          String   @id @default(cuid())
  slug        String   @unique
  name        String
  version     String?
  downloadUrl String
  imageUrl    String   // 软件图标/预览图
  adSlotId    String?  // 关联广告位
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

model AdSlot {
  id        String      @id @default(cuid())
  name      String
  type      String      // banner, insertion, native, splash
  position  String
  width     Int
  height    Int
  isActive  Boolean     @default(true)
  theme     String      @default("auto") // light/dark/auto
  adContents AdContent[]
}

model AdContent {
  id          String   @id @default(cuid())
  slotId      String
  slot        AdSlot   @relation(fields: [slotId], references: [id])
  title       String
  description String?
  imageUrl    String
  targetUrl   String
  ctaText     String   @default("了解更多")
  advertiser  String
  isActive    Boolean  @default(true)

  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
}

model DownloadLog {
  id          String   @id @default(cuid())
  softwareId  String
  software    Software @relation(fields: [softwareId], references: [id])
  ip          String
  userAgent   String?
```
